<script>

let adminUser="admin"
let adminPass="12345"

function login(){

let u=username.value
let p=password.value

if(u==adminUser && p==adminPass){

loginPage.style.display="none"
dashboard.style.display="block"

init()

}else{

alert("Login gagal")

}

}

let staffDB=JSON.parse(localStorage.getItem("staffDB")) || []

let izinDB=JSON.parse(localStorage.getItem("izinDB")) || []

function saveDB(){

localStorage.setItem("staffDB",JSON.stringify(staffDB))
localStorage.setItem("izinDB",JSON.stringify(izinDB))

}

function addStaff(){

let s={

id:Date.now(),
nama:nama.value,
jobdesk:jobdesk.value,
role:role.value

}

staffDB.push(s)

saveDB()

renderStaff()

}

function renderStaff(){

let html=""

staffDB.forEach(s=>{

let totalHari=countToday(s.id)

html+=`

<tr>

<td>${s.nama}</td>

<td>${s.jobdesk}</td>

<td>${s.role}</td>

<td>

<button onclick="izin(${s.id})">

Mulai Izin

</button>

</td>

<td id="timer-${s.id}">-</td>

<td>${totalHari}</td>

</tr>

`

})

staffTable.innerHTML=html

}

function izin(id){

let today=new Date().toDateString()

let total=countToday(id)

if(total>=4){

alert("Maksimal izin 4x hari ini")

return

}

izinDB.push({

staff:id,
start:Date.now(),
tanggal:today

})

saveDB()

}

function countToday(id){

let today=new Date().toDateString()

return izinDB.filter(x=>x.staff==id && x.tanggal==today).length

}

function updateTimer(){

let now=Date.now()

staffDB.forEach(s=>{

let data=izinDB.filter(x=>x.staff==s.id)

if(data.length==0) return

let last=data[data.length-1]

let durasi=Math.floor((now-last.start)/1000)

let sisa=900-durasi

let el=document.getElementById("timer-"+s.id)

if(!el) return

if(sisa>0){

let m=Math.floor(sisa/60)
let d=sisa%60

el.innerHTML=m+":"+d

el.parentElement.className="green"

}else{

let lewat=Math.abs(sisa)

let m=Math.floor(lewat/60)
let d=lewat%60

el.innerHTML="Telat "+m+":"+d

el.parentElement.className="red"

}

})

}

function resetHarian(){

let today=new Date().toDateString()

izinDB=izinDB.filter(x=>x.tanggal==today)

saveDB()

}

function grafik(){

let data={}

izinDB.forEach(x=>{

let staff=staffDB.find(s=>s.id==x.staff)

if(!staff) return

if(!data[staff.nama]) data[staff.nama]=0

data[staff.nama]++

})

let labels=Object.keys(data)
let values=Object.values(data)

new Chart(document.getElementById("chart"),{

type:"bar",

data:{

labels:labels,

datasets:[{

label:"Jumlah Izin",

data:values

}]

}

})

}

function init(){

renderStaff()

grafik()

setInterval(updateTimer,1000)

setInterval(resetHarian,60000)

}
